wavegenCan be TC_WAVE_WAVEGEN_NFRQ, TC_WAVE_WAVEGEN_MFRQ, TC_WAVE_WAVEGEN_NPWM or TC_WAVE_WAVEGEN_MPWM
countdirCan be TC_COUNT_DIRECTION_UP or TC_COUNT_DIRECTION_DOWN
